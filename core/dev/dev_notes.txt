3- "System" command:
	-> echo -n "Hostname: "; /bin/hostname
	-> echo -n "Kernel: "; /bin/uname -a
	-> echo -n "OS: ";  /bin/cat /etc/*-release | /usr/bin/head -n 1
	-> echo -n "Time: "; /bin/date
	-> echo -n "Timezone (UTC): "; /usr/sbin/zdump UTC 
	-> echo -n "Language: ";  echo $LANG
4- "IP" command:
	-> echo -n "Internal IP/subnet: "; ip addr show | grep inet | awk '{print $2}'
	-> echo -n "External IP: "; curl http://ifconfig.me/ip
	-> echo -n "DNS: "; cat /etc/resolv.conf | awk {'print $2'}
	-> echo -n "Gateway (eth0): "; route -n | grep eth0 | awk '{print $2}' | head -n 1
	-> echo -n "DHCP? (eth0): "; grep "BOOTPROTO=dhcp" /etc/sysconfig/network-scripts/ifcfg-eth0; if [[ $? -eq 1 ]]; then echo "No"; else echo "Yes"; fi
	-> echo -n "DHCP Server: "; cat /var/lib/dhcp3/dhclient.leases	
5- "Health" command:
	-> echo -n "Uptime: "; /usr/bin/uptime | /bin/awk '{print $3 ":" $5}' | /usr/bin/tr -d "," | /bin/awk -F ":" '{print $1 " days, " $2 " hours and " $3 " minutes" }'
	-> echo -n "Idle time ($): "; /bin/awk '{print ($1/(60*60*24))/($2/(60*60*24))*100 "%"}' /proc/uptime
	-> echo -n "Users Logged in: "; /usr/bin/w -h | /usr/bin/wc -l 
	-> echo -n "Total Users: "; /usr/bin/wc -l /etc/passwd | /bin/awk '{print $1}'
	-> echo -n "Total Groups: "; /usr/bin/wc -l /etc/group | /bin/awk '{print $1}'
	-> echo -n "CPU Load (1, 5, 15 mins): "; /bin/awk '{print $1 " " $2 " " $3}' /proc/loadavg
	-> echo -n "Memory Load (Used %): "; /usr/bin/free -m | /bin/grep 'buffers/cache' | /bin/awk '{print $3*100/($3+$4)}'
	-> echo -n "Established TCP Connections: "; /bin/netstat -tn | /bin/grep ESTABLISHED | /usr/bin/wc -l | /bin/awk "{print $1}"
	-> echo -n "Listening TCP Services: "; /bin/netstat -atn | /bin/grep LISTEN | /usr/bin/wc -l | /bin/awk "{print $1}"
	-> echo -n "User Processors: "; /bin/awk '{split($4,a,"/"); print a[1];}' /proc/loadavg
	-> echo -n "Total Processors: "; /bin/awk '{split($4,a,"/"); print a[2];}' /proc/loadavg
	-> echo -n "Processors (Total): "; /bin/awk '{split($4,a,"/"); printf a[1] " ("; printf a[2]")"}' /proc/loadavg
6- "History" command:
	-> 	for i in $(cut -d: -f6 /etc/passwd | sort | uniq); do
			if [ -f $i/.bash_history ]; then echo "bash_history: $i"; fi
			if [ -f $i/.nano_history ]; then echo "nano_history: $i"; fi
			if [ -f $i/.atftp_history ]; then echo "atftp_history: $i"; fi
			if [ -f $i/.mysql_history ]; then echo "mysql_history: $i"; fi
			if [ -f $i/.php_history ]; then echo "php_history: $i"; fi
		done
7- "Keys" command:
	-> echo -n "SSL Certs: "; find / -type f -print0 | xargs -0 -I '{}' bash -c 'openssl x509 -in {} -noout > /dev/null 2>&1; [[ $? == '0' ]] && echo "{}"'
	-> echo -n "SSH Public: "; find / -type f -print0 | xargs -0 -I '{}' bash -c '/usr/bin/ssh-keygen -lf {} >/dev/null 2>&1; [[ $? == '0' ]] && echo "{}"'
